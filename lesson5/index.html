function toSixteen(a) {//a= 25
var i = 0, b = [], str = '';
while ((a > 0) && (i<100)){
b[i] = a % 16;//Остаток от / 16, b[0] = 9
switch (b[i]){
case 10: b[i] = 'A'; break;
case 11: b[i] = 'B'; break;
case 12: b[i] = 'C'; break;
case 13: b[i] = 'D'; break;
case 14: b[i] = 'E'; break;
case 15: b[i] = 'F'; break;
}
// a = (a / 16).toFixed(0); не работает
a = Math.floor(a / 16);
str = b[i].toString() + str;
i++;
}
if (i==100){
str = 'зациклилось!!'
}
return str + '<br>';
}
