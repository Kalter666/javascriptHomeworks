 function oneByOne(i) { // Через 1 первый всегда пробел
var str = '';
var PJSharp = false;
while (i > 0){
switch (PJSharp){
case false: str = str + ' '; PJSharp = true; break;
case true: str = str + '#'; PJSharp = false;
}
i--;
}
return str;
}

}

}
